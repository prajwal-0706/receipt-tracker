Examples:

Q: How much did I spend on coffee last month?
SQL: SELECT COALESCE(SUM(li.total_price), 0) AS total_spent FROM line_items li JOIN receipts r ON r.id = li.receipt_id WHERE li.category = 'dining' AND (li.description ILIKE '%coffee%' OR li.description ILIKE '%latte%' OR li.description ILIKE '%espresso%') AND r.receipt_date >= date_trunc('month', current_date - interval '1 month') AND r.receipt_date < date_trunc('month', current_date)

Q: What was my biggest grocery bill in March?
SQL: SELECT r.id, r.merchant, r.receipt_date, r.total FROM receipts r WHERE r.id IN (SELECT receipt_id FROM line_items WHERE category = 'groceries') AND extract(month from r.receipt_date) = 3 AND extract(year from r.receipt_date) = extract(year from current_date) ORDER BY r.total DESC NULLS LAST LIMIT 1

Q: How much did I spend in total this year?
SQL: SELECT COALESCE(SUM(total), 0) AS total_spent, currency FROM receipts WHERE receipt_date >= date_trunc('year', current_date) GROUP BY currency

Q: Show my top 5 merchants by spend.
SQL: SELECT merchant, SUM(total) AS spent FROM receipts WHERE merchant IS NOT NULL GROUP BY merchant ORDER BY spent DESC NULLS LAST LIMIT 5

Q: How many receipts have I uploaded?
SQL: SELECT COUNT(*) AS receipt_count FROM receipts

Q: What's my average dining bill?
SQL: SELECT AVG(r.total) AS avg_bill FROM receipts r WHERE r.id IN (SELECT receipt_id FROM line_items WHERE category = 'dining')

Q: Tell me more about my recent transaction with TechWave Solutions
SQL: SELECT r.id, r.merchant, r.receipt_date, r.total, r.currency, li.description, li.quantity, li.total_price, li.category FROM receipts r LEFT JOIN line_items li ON li.receipt_id = r.id WHERE r.merchant ILIKE '%techwave%' ORDER BY r.receipt_date DESC LIMIT 50

Q: How much have I spent at Walmart?
SQL: SELECT COALESCE(SUM(total), 0) AS total_spent, COUNT(*) AS receipt_count, currency FROM receipts WHERE merchant ILIKE '%walmart%' GROUP BY currency

Q: List my transactions with Costa Coffee
SQL: SELECT id, merchant, receipt_date, total, currency FROM receipts WHERE merchant ILIKE '%costa%' ORDER BY receipt_date DESC

Q: How much did I spend on shopping?

Q: How much did I spend on entertainment?
SQL: SELECT COALESCE(SUM(total_price), 0) AS total_spent, 'entertainment' AS category FROM line_items WHERE category = 'entertainment'

Q: How many items did I buy in total?
SQL: SELECT COALESCE(SUM(quantity), 0) AS total_items FROM line_items

Q: What was the most expensive item I bought?
SQL: SELECT description, quantity, unit_price, total_price FROM line_items ORDER BY total_price DESC NULLS LAST LIMIT 1

Q: What is the cheapest receipt I have?
SQL: SELECT id, merchant, receipt_date, total, currency FROM receipts ORDER BY total ASC NULLS LAST LIMIT 1
